Main Tab Shell Controller (src/app/page.tsx)

"use client";
import React, { useState } from 'react';
import { AnimatePresence, motion } from 'framer-motion';
import BottomNav from '@/components/BottomNav';
import TabDiscover from '@/components/tabs/TabDiscover';
import TabSaved from '@/components/tabs/TabSaved';
import TabMyRuns from '@/components/tabs/TabMyRuns';
import TabLeaderboard from '@/components/tabs/TabLeaderboard';
import TabProfile from '@/components/tabs/TabProfile';

export default function Home() {
  const [activeTab, setActiveTab] = useState(1);

  return (
    <main className="relative mx-auto h-[100dvh] w-full max-w-md overflow-hidden bg-[#050505] text-white border-x border-white/10 shadow-2xl flex flex-col">
      <div className="flex-1 overflow-y-auto pb-24 scrollbar-none">
        <AnimatePresence mode="wait">
          <motion.div
            key={activeTab}
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -10 }}
            transition={{ duration: 0.2 }}
            className="h-full"
          >
            {activeTab === 1 && <TabDiscover />}
            {activeTab === 2 && <TabSaved />}
            {activeTab === 3 && <TabMyRuns />}
            {activeTab === 4 && <TabLeaderboard />}
            {activeTab === 5 && <TabProfile />}
          </motion.div>
        </AnimatePresence>
      </div>
      <BottomNav activeTab={activeTab} setActiveTab={setActiveTab} />
    </main>
  );
}
